STL

stl是C++程序库，我们只需要了解它的 1 容器 2 函数（内置算法） 3 迭代器，不同的stl容器有不同的存储方法，包含不同的函数，做题时根据输入，输出，以及所需的操作来选择

常用stl容器：1 vector 2 stack 3 queue 4 map 5 set 6 list

以vector举例

我们用一个容器应该知道它的特性 明白两点 它的特性 以及如何使用它

​ vector其实就是类似于数组 ，不过多了很多内置函数方便你处理

​ 如何使用它：我给出一下实例 建议自己边敲边理解

你只需要知道 怎么定义 怎么输入 怎么输出 以及有哪些可用函数

#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	
	#1 初始化vector  创建了两个vector容器 a,b  
	vector<int>a,b;
	
	#2 输入 
	for(int i=0;i<n;i++){
		
		int shuru;
		cin>>shuru;
		#根据英文理解就是从后面输入元素 0<-1<-2 
		a.push_back(shuru);
	}
	
	#3 输出（访问） it 是迭代器 也就是 a的 指针  用于访问vector元素 
	for(vector<int>::iterator it=a.begin();it<a.end();it++){
		#联想指针的用法 it是它的地址 
		cout<<*it;
	}
		
	#4 vector 的 函数
		
	#增 
		#assian用于输入 
			#1将a的前0-2号元素复制给b 
			b.assign(a.begin(),a.begin()+3);
			#2给b初始化4个为2的元素 
			b.assign(4,2);
			
		#a.insert() 插入
		
		a.insert(a.begin()+1,3);//在第一个元素后面插入3 
		 
		int num=0; 
		a.insert(a.begin()+1,3,num);//在第一个元素后面插入3个num 
		
		a.insert(a.begin()+1,b.begin()+2,b.begin()+4);//在a的第一个元素后面插入b的2-4号元素值 
		
		#增加100个容器容量 
		a.reserve(100);
		 
		#a.resize() #增加10个元素 
		a.resize(10);
		# 增加10个元素  初值为2 
		a.resize(10,2);
		
	#查 
		# a.empty 用于判空 
		if(a.empty()) {
			cout<<"a is empty"； 
		}
		
		#a.size a的元素个数 
		int lenth=a.size(); 
		
		#a.capacity() a能容纳多少个元素 
		int room=a.capacity();
		
		#访问第一个元素 
		int first=a.front();
		
		#访问最后一个元素 
		int last =a.back();
		
		 
		
	#删 
		#a.pop_back() 根据英语理解就是删除最后一个元素 
		a.pop_back();
	
		#a.erase() 删除0-2这个三个元素 
		a.erase(a.begin()+1,a.begin()+3); 
		
		#清空a的元素 
		a.clear();
	
	#改
		#a b 交互 
		a.swap(b);
		
		
	#常用函数
	  #include<algorithm>//用内置算法需要这个库
 //对a中的从a.begin()（包括它）到a.end()（不包括它）的元素进行从小到大排列
 sort(a.begin(),a.end());
 //对a中的从a.begin()（包括它）到a.end()（不包括它）的元素倒置，但不排列，如a中元素为1,3,2,4,倒置后为4,2,3,1
 reverse(a.begin(),a.end());
  //把a中的从a.begin()（包括它）到a.end()（不包括它）的元素复制到b中，从b.begin()+1的位置（包括它）开始复制，覆盖掉原有元素
 copy(a.begin(),a.end(),b.begin()+1);
 //在a中的从a.begin()（包括它）到a.end()（不包括它）的元素中查找10，若存在返回其在向量中的位置
  find(a.begin(),a.end(),10);

		 
}

接下来我们进行一个实战

http://acm.swust.edu.cn/#/problem/254/496

#include<bits/stdc++.h>//这里直接用万能头文件 vector的库来自于#include<vector>
using namespace std;

int main(){
	//这个题需要进行反复的数组翻转所以用vector和reverse函数非常合适 
	vector<int>a;


	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		int shuru;
		cin>>shuru;
		a.push_back(shuru);
	} 
	
	//c.begin() 返回一个迭代器，它指向容器c的第一个元素
	//c.end() 返回一个迭代器，它指向容器c的最后一个元素的下一个位置
	//c.rbegin() 返回一个逆序迭代器，它指向容器c的最后一个元素
	//c.rend() 返回一个逆序迭代器，它指向容器c的第一个元素前面的位置	
	//!!!begin 和 end 一起  rbegin 和 rend 一起 不能混用 这里我使用的 iterator 它和 reverse_iterator的规律是相反的 
	
	vector<int>::iterator it=a.end()-1;

	//这个题要执行的两个操作 一个就是从右向左找到最大的 然后把它反转到最左边， 一个就是把它翻转到最右边
	//直到 整个vector呈升序
	int ji=0;//记录翻转次数
	
	//第一层循环，每次循环确保右边的一位为最大值，第二层循环去寻找最大值 ,然后进行翻转 
	for(;it != a.begin();it--) {
		int max=*it;
		vector<int>::iterator p=it;
		vector<int>::iterator r=it;
		int flag=0;
		
		
		for(vector<int>::iterator it1=it-1;it1 != a.begin()-1;it1--){

			 if(*it1>max){
			 	p=it1;
			 	max=*it1;
			 	flag=1;
			 }
		

		}//找到最大值max以及它的位置p 
		
		if(flag==0){
			continue;//说明最后一个就是最大值不用翻转 
		}		
		
		if(p==a.begin()){//情况一 
			ji+=1;
			reverse(a.begin(),r+1);

		}
		else{//情况二 
			ji+=2;
			reverse(a.begin(),p+1);

			reverse(a.begin(),r+1);

		} 


	}
	cout<<ji<<endl; 

	
}

ok 接下来 我们来介绍两个简单的

stack 即是栈 queue 就是队列

对于栈来说 记住3点

  1. stack 先进后出
  2. #include导入库 stacksta 定义
  3. sta.push()进 sta.pop()出 sta.top()查看栈顶 sta.empty()查空 sta.size()查看大小

对于队列来说 同上：

  1. queue先进先出
  2. #include queueque 定义
  3. que.push() que.pop() que.empty() que.size() 这些用法同上 就是访问不同que.front() que.back()

stack和queue 还有更高级的用法 后面用到再讲

#include<iostream>
#include<stack>
#include<algorithm>
using namespace std;
int main(){
	stack<char>sta;
	char shuru;
	string a;
	cin>>a;
	for(int i=0;i<a.length();i++){
		shuru=a[i];
		if(shuru='"'){
			continue;
		}
		if(sta.empty()||shuru=='('||shuru=='{'||shuru=='[')
			sta.push(shuru);
		else{
			
			if((shuru==')'&&sta.top()!='(')||(shuru==']'&&sta.top()!='[')||(shuru=='}'&&sta.top()!='{')){
				cout<<"false\n";
				exit(0);
			}
			sta.pop();
		}		
	} 
	if(a.length()%2==0)
	cout<<"true\n";
	else{
	cout<<"false\n";		
	}
}

牛客通过答案

class Solution {
public:
    /**
     * 
     * @param s string字符串 
     * @return bool布尔型
     */
    bool isValid(string s) {
        // write code here
        stack<char>st;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='('||s[i]=='{'||s[i]=='[')
                st.push(s[i]);
            else if(s[i]==')')
            {
                if(st.empty()==true||st.top()!='(')
                {
                    return false;
                }
                    st.pop();
            }
            else if(s[i]==']')
            {
                if(st.empty()==true||st.top()!='[')
                {
                    return false;
                }
                    st.pop();
            }
            else
            {
                if(st.empty()==true||st.top()!='{')
                {
                    return false;
                }
                    st.pop();
            }
        }
        if(st.empty()==true)
            return true;
        else
            return false;
    }
};
